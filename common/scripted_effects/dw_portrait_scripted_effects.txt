# Cycles through custom NPC images
dw_cycle_custom_image = {
    if = {
        limit = { has_character_flag = dw_npc_harley }
        if = {
            limit = { trait = harley01_pic }
            remove_trait = harley01_pic
            add_trait = harley02_pic
        }
        else_if = {
            limit = { trait = harley02_pic }
            remove_trait = harley02_pic
            add_trait = harley03_pic
        }
        else_if = {
            limit = { trait = harley03_pic }
            remove_trait = harley03_pic
            add_trait = harley04_pic
        }
        else_if = {
            limit = { trait = harley04_pic }
            remove_trait = harley04_pic
            add_trait = harley01_pic
        }
        else_if = { # For those that want to make her young looking again
            limit = { trait = harley05_pic }
            remove_trait = harley05_pic
            set_character_flag = dw_harley_look_young
            add_trait = harley01_pic
        }
    }
    if = {
        limit = { has_character_flag = dw_npc_batgirl }
        if = {
            limit = { trait = batgirl01_pic }
            remove_trait = batgirl01_pic
            add_trait = batgirl02_pic
        }
        else_if = {
            limit = { trait = batgirl02_pic }
            remove_trait = batgirl02_pic
