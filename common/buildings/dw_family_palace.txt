### Family Palace buildings for Dark World Reborn
## Used by Merchant Republics

family_palace = {
	dw_lilith_temple = {
		potential = {
			FROM = {
				OR = {
					trait = lilith
					trait = incubus
					has_character_flag = dw_build_chapel_lilith
				}
			}
		}
		desc = dw_lilith_temple_desc
		add_number_to_name = no # Not necessary with only one version
		trigger = { FROM = { TECH_CITY_CONSTRUCTION = 0 } } # No special technology needs for a basic shrine
		gold_cost = 100
		build_time = 240
		ai_creation_factor = 50
		extra_tech_building_start = 10 # Should prevent any from being pre-built at game start
		convert_to_castle = dw_chapel_lilith
		#modifiers provided by the building:
		liege_piety = 0.5 # 0.5 monthly increase - Lilith is pleased
	}
	dw_lilith_shrine_outdoors_merchant = {
		potential = {
			FROM = { # This is unique to the Chosen of Lilith Society.
				has_character_flag = dw_build_chapel_lilith
			}
		}
		desc = dw_lilith_shrine_outdoors_desc
		add_number_to_name = no # Not necessary with only one version
		trigger = { FROM = { TECH_CITY_CONSTRUCTION = 0 } } # No special technology needs for a basic shrine
		gold_cost = 100 # A cost to cover collecting the items for the shrine
		build_time = 365
		ai_creation_factor = 50
		convert_to_castle = dw_lilith_shrine_outdoors_feudal
		extra_tech_building_start = 10 # Should prevent any from being pre-built at game start
		#modifiers provided by the building:
		liege_piety = 0.5 # 0.5 monthly increase - Lilith is pleased
	}
}
