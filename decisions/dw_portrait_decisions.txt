targeted_decisions = {
    ## Remove a custom portrait
	dw_remove_custom_portrait = {
		filter = all
		ai_target_filter = none
		ai = no # player only allowed right now
		from_potential = {
			# We have no specific requirements
		}
		potential = {
			# They should already have a portrait
            OR = {
                has_custom_dw_pic = yes
                has_custom_portrait = yes
            }
		}
		allow = {
			always = yes
		}
		effect = {
			remove_custom_portrait = yes
            remove_custom_dw_pic = yes
		}
		ai_will_do = {
			factor = 0 # we don't want the ai removing portraits
		}
	}
    ## Apply a random trap image
	dw_apply_random_trap_image = {
		filter = court_including_me # Just in case a player wants to alter those inside their court
		ai_target_filter = self # We only want the ai to target themselves, if they're ever allowed to do it.
		ai = no # player only allowed right now
		from_potential = {
			# No specific requirements to use
		}
		potential = {
			# The target must be a trap
			trait = trap
			# And they should not already have a portrait
			NOT = { has_custom_portrait = yes }
		}
		allow = {
			always = yes
		}
		effect = {
            assign_random_trap_portrait = yes
		}
		ai_will_do = {
			factor = 100 # When the ai is allowed to do so, it will
		}
	}
    dw_harley_gets_old = { # Harley changes her portrait once she hits 50
		is_high_prio = yes # Will show an alert once decision is enabled
		potential = {
			has_character_flag = dw_npc_harley
			NOT = { trait = harley05_pic }
		}
		allow = {
            FROM = { character = yes }
			age = 50
			NOT = { has_character_flag = dw_harley_look_young }
		}
		effect = {
			remove_trait = harley01_pic
			remove_trait = harley02_pic
			remove_trait = harley03_pic
			remove_trait = harley04_pic
			add_trait = harley05_pic
		}
		ai_will_do = {
			factor = 1 # 100%
		}
	}
    dw_dash_matures = { # Dash becomes an adult
		is_high_prio = yes # Will show an alert once decision is enabled
		potential = {
			has_character_flag = dw_npc_dashparr
		}
		allow = {
            FROM = { character = yes }
			is_adult = yes
			NOT = { trait = dashparr01_pic }
		}
		effect = {
			remove_trait = dashparr02_pic
			add_trait = dashparr01_pic
		}
		ai_will_do = {
			factor = 1 # 100%
		}
	}
    dw_give_me_random_tentacle_image = {
		filter = self
		ai_target_filter = self
		ai_check_interval = 12 # No need to check for this more than once a year
		from_potential = {
			trait = tentacle # Only for those who have the trait
		}
		potential = {
			character = FROM # We want to target ourselves
			trait = tentacle # Only for those who have the trait
		}
		allow = {
			OR = {
				has_character_modifier = dw_out_of_the_shadows # Our presence must be known by others for a visible change
				has_character_modifier = dw_enemy_of_mankind
			}
			NOT = { has_character_flag = dw_dont_give_custom_tentacle_pic } # For players who've decided no custom tentacle pic, ever, for them
		}
		effect = {
			assign_random_dw_tentacle_pic = yes
		}
		ai_will_do = {
			factor = 1.0 #100% likely - assuming they meet the requirements
			modifier = {
				factor = 0
				has_custom_dw_pic = yes
			}
		}
	}
    # Hyena-Kin characters can give themselves a random hyena based image - if they've come out
	dw_give_me_random_hyena_image = {
		filter = self
		ai_target_filter = self
		ai_check_interval = 12 # No need to check for this more than once a year
		potential = {
			character = FROM # We want to target ourselves
			trait = hyena_kin # Only for those who have the trait
		}
		allow = {
			OR = {
				has_character_modifier = dw_out_of_their_skin # Our presence must be known by others for a visible change
				has_character_modifier = dw_enemy_of_mankind
			}
			NOT = { has_character_flag = dw_dont_give_custom_hyena_pic } # For players who've decided no custom tentacle pic, ever, for them
		}
		effect = {
			assign_random_dw_hyena_pic = yes
		}
		ai_will_do = {
			factor = 1.0 #100% likely - assuming they meet the requirements
			modifier = {
				factor = 0
				has_custom_dw_pic = yes
			}
		}
	}
    # Player can flag an npc to stop taking a random tentacle image
	dw_no_more_custom_tentacle_pic = {
		filter = all
		ai_target_filter = self # ai shouldn't be able to do this
		from_potential = {
			ai = no # Player only
		}
		potential = {
			has_custom_dw_pic = yes
		}
		allow = {
			trait = tentacle
		}
		effect = {
			remove_custom_dw_pic = yes
			set_character_flag = dw_dont_give_custom_tentacle_pic
		}
		ai_will_do = {
			factor = 0 # ai should never use this
		}
	}
	# Player can flag an npc to stop taking a random hyena_kin image
	dw_no_more_custom_hyena_pic = {
		filter = all
		ai_target_filter = self # ai shouldn't be able to do this
		from_potential = {
			ai = no # Player only
		}
		potential = {
			has_custom_dw_pic = yes
		}
		allow = {
			trait = hyena_kin
		}
		effect = {
			remove_custom_dw_pic = yes
			set_character_flag = dw_dont_give_custom_hyena_pic
		}
		ai_will_do = {
			factor = 0 # ai should never use this
		}
	}
    # Cycle through Custom NPc Image
	dw_cycle_custom_npc_image = {
		filter = all
		ai_target_filter = self
		from_potential = {
			ai = no # The player should be the only one looking to change images
		}
		potential = {
			does_the_npc_have_multiple_image_options = yes
		}
		allow = {
			custom_tooltip = {
				text = dw_multiple_image_tooltip
				does_the_npc_have_multiple_image_options = yes
			}
		}
		effect = {
			dw_cycle_custom_image = yes
		}
		ai_will_do = { factor = 0 } # ai shouldn't be using this one
	}
}
