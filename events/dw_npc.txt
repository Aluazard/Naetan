	window = "BiggerEventWindow"
	picture = harley_quinn
	title = dw_hello_harley_title
	is_triggered_only = yes
	immediate = {
		dw_create_harley = yes
	}
	option = { # Keep her
		name = dw_npc100a
		hidden_tooltip = { character_event = { id = dw_npc.101 }}
	}
	option = { # banish
		name = dw_npc100b
		event_target:dw_harley_recruit = {
			banish = yes
		}
	}
	option = { # Never bother me again
		name = dw_npc100c
		hidden_tooltip = {
			set_global_flag = dw_npc_ignore_harley
			clr_global_flag = dw_npc_harley_alive
			event_target:dw_harley_recruit = {
				death = { death_reason = death_vanished }
			}
		}
	}
}
