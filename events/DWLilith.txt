# .10 The dream (hidden routing event)
character_event = {
	id = DWLilith.10
	hide_window = yes # Nothing to see here
	is_triggered_only = yes # Triggered by decision
	ai = no
	trigger = {
		has_heaven_hell_content = yes
		has_lilith_start = yes
	}
	immediate = {
		if = { # Male event
			limit = {
				is_female = no
			}
			narrative_event = { id = DWLilith.11 }
			break = yes # Takes us out of the immediate block
		}
		# Female event
		narrative_event = { id = DWLilith.15 }
	}
}

			}
			10 = { # target develops a stutter
				modifier = {
					factor = 2 # For those who are masters of intrigue, the horrors are that much more terrifying.
					intrigue = 15
				}
				FROM = { add_trait = stutter }
			}
			10 = { # target becomes slow
				modifier = {
					factor = 2 # For those who are masters of intrigue, the horrors are that much more terrifying.
					intrigue = 15
				}
				FROM = { add_trait = slow }
			}
			5 = { # target becomes an imbecile

					intrigue = 15
				}

			}

						death_reason = death_natural

		}

		name = DWLilith160A
		piety = -50
		FROM = {
			opinion = {
				modifier = dw_opinion_incubus_intimidation
				who = ROOT
			}
		}
	}
	option = { # Break their resolve
		trigger = {
			FROM = { NOT = { trait = craven }} # If the target is already a coward, there is no reason to display this option
		}
		name = DWLilith160B
		piety = -50
		if = { # If they're not brave, we render them a coward
			limit = {
				FROM = { NOT = { trait = brave }}
			}
			FROM = { add_trait = craven }
		}
		if = { # If they're brave, we break it down
			limit = {
				FROM = { trait = brave }
			}
			FROM = { remove_trait = brave }
		}
	}
	option = { # If the ai selected the player
		trigger = {
			ai = yes
			FROM = { ai = no }
		}
		name = DWLilithDoh
